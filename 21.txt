script
    const { createApp, ref, computed } = Vue

    createApp({
        setup() {
            const activeTab = ref('itinerary')
            
            const itinerary = ref([
                { time '0900', activity '清邁古城散步', location '塔佩門' },
                { time '1230', activity '泰式泰式午餐', location 'Blue Noodle' },
                { time '1500', activity '放鬆 SPA', location 'Fah Lanna' }
            ])

            const expenses = ref([
                { id 1, item '機票', amount 12000 },
                { id 2, item '午餐', amount 350 },
                { id 3, item '按摩', amount 800 }
            ])

            const shoppingList = ref([
                { id 1, name '泰式奶茶包', done false },
                { id 2, name '芒果乾', done true },
                { id 3, name '香氛精油', done false }
            ])

            const totalExpenses = computed(() = {
                return expenses.value.reduce((sum, item) = sum + item.amount, 0)
            })

            return { activeTab, itinerary, expenses, shoppingList, totalExpenses }
        }
    }).mount('#app')
script
